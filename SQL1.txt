CREATE TABLE STUDENT(ROLLNO NUMBER(10) PRIMARY KEY ,
                     NAMES VARCHAR(30) NOT NULL,
                     CITY  VARCHAR (30) DEFAULT ‘RAJKOT’
                     MOBILE NUMBER(10) UNIQUE CHECK(LENGTH(MOBILE)=10)
)

CREATE TABLE CUSTOMER(CUST_NO VARCHAR(5) PRIMARY KEY CONSTRAINT chck_cust_no startwith_c CHECK(CUST_NO LIKE 'C____'),
                     CNAME VARCHAR(30) NOT NULL CONSTRAINT chck_cname_start_A_P CHECK(SUBSTR(CNAME,1,1) BETWEEN 'A' AND 'P'),
                     EMAILID VARCHAR(30) UNIQUE CONSTRAINT chck_email_has_@ CHECK(EMAILID LIKE '___%@__%.___%')
)
